'use client';
 
import { useEffect, useMemo, useRef, useState, useCallback } from 'react';
import { supportAPI } from '@/lib/api';
import { useToast } from '@/hooks/use-toast';
import { useTypingIndicator } from '@/hooks/useTypingIndicator';
import { useChatPagination } from '@/hooks/useChatPagination';
import { useChatDrafts } from '@/hooks/useChatDrafts';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Avatar, AvatarFallback } from '@/components/ui/avatar';
import { Badge } from '@/components/ui/badge';
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs';
import { Checkbox } from '@/components/ui/checkbox';
import {
  Search,
  Send,
  Paperclip,
  Download,
  Upload,
  X,
  Check,
  CheckCheck,
  Clock,
  AlertCircle,
  Filter,
  MessageCircle,
} from 'lucide-react';
import { motion } from 'framer-motion';
 
const API_BASE_URL = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000';
 
interface Message {
  id: string;
  ticket_id: string;
  sender_id: string;
  sender_role: string;
  message: string;
  attachments?: Array<{ filename: string; url: string; size: number; type: string }>;
  is_read: boolean;
  is_internal: boolean;
  created_at: string;
  read_at?: string;
}
 
interface Ticket {
  id: string;
  subject: string;
  category: string;
  status: string;
  priority: string;
  created_at: string;
  updated_at: string;
  unread_count?: number;
  last_message?: Message;
  creator?: {
    id: string;
    email: string;
    full_name: string;
    company_name: string;
  };
}
 
const PRIORITY_COLORS: Record<string, string> = {
  low: 'bg-gray-100 text-gray-800 dark:bg-gray-800 dark:text-gray-200',
  medium: 'bg-blue-100 text-blue-800 dark:bg-blue-900 dark:text-blue-200',
  high: 'bg-orange-100 text-orange-800 dark:bg-orange-900 dark:text-orange-200',
  urgent: 'bg-red-100 text-red-800 dark:bg-red-900 dark:text-red-200',
};
 
const CATEGORY_ICONS: Record<string, string> = {
  technical: '',
  billing: '',
  bug: '',
  feature: '',
  data: '',
  dashboard: '',
  other: '',
};
 
export default function AdminSupportPageEnhanced() {
  const { toast } = useToast();
  const [tickets, setTickets] = useState<Ticket[]>([]);
  const [selected, setSelected] = useState<Ticket | null>(null);
  const [messages, setMessages] = useState<Message[]>([]);
  const [seller, setSeller] = useState<any | null>(null);
  const [filters, setFilters] = useState({ status: '', priority: '', category: '', q: '' });
  const [uploading, setUploading] = useState(false);
  const [uploadedFiles, setUploadedFiles] = useState<any[]>([]);
  const [isInternal, setIsInternal] = useState(false);
  const [searchQuery, setSearchQuery] = useState('');
  const [showFilters, setShowFilters] = useState(false);
 
  const listRef = useRef<HTMLDivElement>(null);
  const fileInputRef = useRef<HTMLInputElement>(null);
  const prevSellerMsgCount = useRef(0);
 
  const { draft, saveDraft, clearDraft } = useChatDrafts(selected?.id || null);
  const { remoteTyping, notifyTyping, stopTyping } = useTypingIndicator(selected?.id || null);
  const { offset, hasMore, loadMore, reset, updateHasMore } = useChatPagination();
 
  const [reply, setReply] = useState('');
 
  useEffect(() => {
    setReply(draft);
  }, [draft]);
 
  // Handle file download with authentication
  const handleFileDownload = async (url: string, filename: string) => {
    try {
      const fileUrl = url.startsWith('http') ? url : `${API_BASE_URL}${url}`;
      const urlObj = new URL(fileUrl);
      const filenameFromPath = urlObj.pathname.split('/').pop() || filename;
 
      const response = await supportAPI.downloadFile(filenameFromPath);
      const blob = new Blob([response.data]);
      const downloadUrl = window.URL.createObjectURL(blob);
      const link = document.createElement('a');
      link.href = downloadUrl;
      link.download = filename;
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      window.URL.revokeObjectURL(downloadUrl);
    } catch (error: any) {
      toast({
        title: 'Download failed',
        description: error?.response?.data?.detail || 'Could not download file',
        variant: 'destructive',
      });
    }
  };
 
  // Fetch tickets
  useEffect(() => {
    let alive = true;
    const load = async () => {
      try {
        const res = await supportAPI.listAllTickets({
          status: filters.status || undefined,
          priority: filters.priority || undefined,
          category: filters.category || undefined,
          q: filters.q || undefined,
        });
        if (!alive) return;
        setTickets(res.data || []);
      } catch (e: any) {
        if (alive && e?.response?.status !== 429) {
          console.error('Failed to load tickets:', e?.message);
        }
      }
    };
    load();
    const id = setInterval(load, 10000); // Increased from 5s to 10s
    return () => {
      alive = false;
      clearInterval(id);
    };
  }, [filters]);
 
  // Fetch messages
  useEffect(() => {
    if (!selected) return;
    let alive = true;
    const load = async () => {
      try {
        const res = await supportAPI.getTicketAdmin(selected.id, {
          limit: 50,
          offset,
          include_internal: true,
        });
        if (!alive) return;
 
        const msgs = res.data?.messages || [];
        const sellerMsgs = msgs.filter((m: Message) => m.sender_role === 'seller' && !m.is_read);
 
        if (sellerMsgs.length > prevSellerMsgCount.current) {
          toast({ title: 'New Message', description: 'Seller sent a new message.' });
 
          // Auto-mark as read
          try {
            await supportAPI.markRead(selected.id);
          } catch (e) {
            // Ignore mark-read errors
          }
        }
        prevSellerMsgCount.current = sellerMsgs.length;
 
        if (offset === 0) {
          setMessages(msgs);
        } else {
          setMessages((prev) => [...msgs, ...prev]);
        }
 
        setSeller(res.data?.seller || null);
        updateHasMore(res.data?.has_more || false);
 
        if (offset === 0 && listRef.current) {
          setTimeout(() => {
            if (listRef.current) {
              listRef.current.scrollTop = listRef.current.scrollHeight;
            }
          }, 100);
        }
      } catch (e: any) {
        if (alive && e?.response?.status !== 429) {
          console.error('Failed to load messages:', e?.message);
        }
      }
    };
    load();
    const id = setInterval(load, 10000); // Increased from 5s to 10s
    return () => {
      alive = false;
      clearInterval(id);
    };
  }, [selected, offset, toast, updateHasMore]);
 
  const unreadCount = useMemo(
    () => tickets.filter((t) => t.unread_count && t.unread_count > 0).length,
    [tickets]
  );
 
  const filteredMessages = useMemo(() => {
    if (!searchQuery.trim()) return messages;
    const query = searchQuery.toLowerCase();
    return messages.filter((m) => m.message.toLowerCase().includes(query));
  }, [messages, searchQuery]);
 
  const sendReply = async () => {
    if (!selected || !reply.trim()) return;
    stopTyping();
 
    await supportAPI.replyTicketAdmin(selected.id, {
      message: reply,
      attachments: uploadedFiles,
      is_internal: isInternal,
    });
 
    setReply('');
    clearDraft();
    setUploadedFiles([]);
    setIsInternal(false);
 
    const res = await supportAPI.getTicketAdmin(selected.id, { limit: 50, offset: 0 });
    setMessages(res.data?.messages || []);
    reset();
 
    toast({
      title: isInternal ? 'Internal note added' : 'Reply sent',
      description: isInternal ? 'Only visible to admins' : 'Seller will see your message.',
    });
  };
 
  const updateStatus = async (status: string) => {
    if (!selected) return;
    await supportAPI.updateStatus(selected.id, { status });
    setSelected({ ...selected, status });
    toast({ title: 'Status updated', description: `Ticket marked as ${status}` });
  };
 
  const updatePriority = async (priority: string) => {
    if (!selected) return;
    await supportAPI.updatePriority(selected.id, { priority });
    setSelected({ ...selected, priority });
    toast({ title: 'Priority updated', description: `Set to ${priority}` });
  };
 
  const handleFileUpload = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file || !selected) return;
 
    if (file.size > 2 * 1024 * 1024) {
      toast({ title: 'File too large', description: 'Max size is 2MB', variant: 'destructive' });
      return;
    }
 
    setUploading(true);
    try {
      const res = await supportAPI.uploadFile(selected.id, file);
      setUploadedFiles((prev) => [...prev, res.data]);
      toast({ title: 'File uploaded', description: file.name });
    } catch (e: any) {
      toast({
        title: 'Upload failed',
        description: e.response?.data?.detail || 'Error',
        variant: 'destructive',
      });
    } finally {
      setUploading(false);
    }
  };
 
  const handleReplyChange = (text: string) => {
    setReply(text);
    saveDraft(text);
    if (text.trim() && !isInternal) {
      notifyTyping();
    } else {
      stopTyping();
    }
  };
 
  const handleScroll = useCallback(() => {
    if (!listRef.current) return;
    const { scrollTop } = listRef.current;
 
    if (scrollTop < 50 && hasMore) {
      loadMore();
    }
  }, [hasMore, loadMore]);
 
  return (
    <div className="flex h-[calc(100vh-7rem)] gap-4">
      {/* Left: Ticket List */}
      <div className="w-2/5 border rounded-lg bg-white dark:bg-gray-900 flex flex-col">
        {/* Filters */}
        <div className="p-3 border-b space-y-2">
          <div className="flex gap-2">
            <div className="relative flex-1">
              <Search className="absolute left-2 top-2.5 h-4 w-4 text-gray-400" />
              <Input
                className="pl-8"
                placeholder="Search tickets..."
                value={filters.q}
                onChange={(e) => setFilters({ ...filters, q: e.target.value })}
              />
            </div>
            <Button
              variant="outline"
              size="icon"
              onClick={() => setShowFilters(!showFilters)}
            >
              <Filter className="h-4 w-4" />
            </Button>
          </div>
 
          {showFilters && (
            <div className="grid grid-cols-3 gap-2">
              <select
                className="px-2 py-2 rounded border bg-transparent text-sm"
                value={filters.status}
                onChange={(e) => setFilters({ ...filters, status: e.target.value })}
              >
                <option value="">All Status</option>
                <option value="open">Open</option>
                <option value="waiting">Waiting</option>
                <option value="active">Active</option>
                <option value="closed">Closed</option>
              </select>
              <select
                className="px-2 py-2 rounded border bg-transparent text-sm"
                value={filters.priority}
                onChange={(e) => setFilters({ ...filters, priority: e.target.value })}
              >
                <option value="">All Priority</option>
                <option value="low">Low</option>
                <option value="medium">Medium</option>
                <option value="high">High</option>
                <option value="urgent">Urgent</option>
              </select>
              <select
                className="px-2 py-2 rounded border bg-transparent text-sm"
                value={filters.category}
                onChange={(e) => setFilters({ ...filters, category: e.target.value })}
              >
                <option value="">All Categories</option>
                <option value="technical">Technical</option>
                <option value="billing">Billing</option>
                <option value="bug">Bug</option>
                <option value="feature">Feature</option>
                <option value="data">Data</option>
                <option value="dashboard">Dashboard</option>
                <option value="other">Other</option>
              </select>
            </div>
          )}
 
          <div className="flex items-center justify-between text-xs text-gray-500">
            <span>
              {tickets.length} ticket{tickets.length !== 1 ? 's' : ''}
            </span>
            {unreadCount > 0 && (
              <Badge variant="destructive" className="text-xs">
                {unreadCount} unread
              </Badge>
            )}
          </div>
        </div>
 
        {/* Ticket List */}
        <div className="flex-1 overflow-auto divide-y">
          {tickets.map((t) => (
            <button
              key={t.id}
              className={`w-full text-left px-3 py-3 hover:bg-gray-50 dark:hover:bg-gray-800 transition-colors ${selected?.id === t.id ? 'bg-blue-50 dark:bg-blue-900/20 border-l-4 border-blue-500' : ''
                }`}
              onClick={() => {
                setSelected(t);
                reset();
                setSearchQuery('');
              }}
            >
              <div className="flex items-start justify-between gap-2">
                <div className="flex-1 min-w-0">
                  <div className="flex items-center gap-2 mb-1">
                    <span className="text-lg">{CATEGORY_ICONS[t.category] || ''}</span>
                    <span className="font-medium truncate">{t.subject}</span>
                  </div>
                  {t.creator && (
                    <div className="text-xs text-gray-600 dark:text-gray-400 mb-1 truncate">
                      {t.creator.full_name}  {t.creator.company_name}
                    </div>
                  )}
                  <div className="flex items-center gap-2 flex-wrap">
                    <Badge className={`text-xs ${PRIORITY_COLORS[t.priority]}`}>
                      {t.priority}
                    </Badge>
                    <Badge variant="outline" className="text-xs">
                      {t.status}
                    </Badge>
                    {t.unread_count && t.unread_count > 0 && (
                      <Badge variant="destructive" className="text-xs">
                        {t.unread_count} new
                      </Badge>
                    )}
                  </div>
                </div>
                <div className="text-xs text-gray-500 flex-shrink-0">
                  <Clock className="h-3 w-3 inline mr-1" />
                  {new Date(t.updated_at).toLocaleDateString()}
                </div>
              </div>
            </button>
          ))}
        </div>
      </div>
 
      {/* Right: Conversation */}
      <div className="flex-1 border rounded-lg bg-white dark:bg-gray-900 flex flex-col">
        {selected ? (
          <>
            {/* Header */}
            <div className="p-3 border-b space-y-2">
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-3">
                  <Avatar className="h-10 w-10 bg-purple-500 text-white">
                    <AvatarFallback>
                      {(seller?.company_name || selected.creator?.company_name)?.charAt(0) || 'S'}
                    </AvatarFallback>
                  </Avatar>
                  <div>
                    <div className="font-semibold">{selected.subject}</div>
                    <div className="text-xs text-gray-500">
                      {seller?.full_name || selected.creator?.full_name}  {seller?.company_name || selected.creator?.company_name}
                    </div>
                    <div className="text-xs text-gray-400">
                      {seller?.email || selected.creator?.email}
                    </div>
                  </div>
                </div>
                <div className="flex gap-2">
                  <select
                    className="px-2 py-1 rounded border bg-transparent text-sm"
                    value={selected.status}
                    onChange={(e) => updateStatus(e.target.value)}
                  >
                    <option value="open">Open</option>
                    <option value="waiting">Waiting</option>
                    <option value="active">Active</option>
                    <option value="closed">Closed</option>
                  </select>
                  <select
                    className="px-2 py-1 rounded border bg-transparent text-sm"
                    value={selected.priority}
                    onChange={(e) => updatePriority(e.target.value)}
                  >
                    <option value="low">Low</option>
                    <option value="medium">Medium</option>
                    <option value="high">High</option>
                    <option value="urgent">Urgent</option>
                  </select>
                </div>
              </div>
 
              {/* Search in conversation */}
              <div className="relative">
                <Search className="absolute left-2 top-2.5 h-4 w-4 text-gray-400" />
                <Input
                  className="pl-8"
                  placeholder="Search in conversation..."
                  value={searchQuery}
                  onChange={(e) => setSearchQuery(e.target.value)}
                />
              </div>
            </div>
 
            {/* Messages */}
            <div
              ref={listRef}
              onScroll={handleScroll}
              className="flex-1 overflow-y-auto p-4 space-y-4"
            >
              {hasMore && (
                <div className="text-center">
                  <Button variant="ghost" size="sm" onClick={loadMore}>
                    Load older messages
                  </Button>
                </div>
              )}
 
              {filteredMessages.map((m) => {
                const isAdmin = m.sender_role === 'admin';
                const highlight = searchQuery && m.message.toLowerCase().includes(searchQuery.toLowerCase());
                const senderName = isAdmin ? 'You (Admin)' : 'Seller';
 
                return (
                  <motion.div
                    key={m.id}
                    initial={{ opacity: 0, y: 10 }}
                    animate={{ opacity: 1, y: 0 }}
                    transition={{ duration: 0.2 }}
                    className="flex gap-3 items-start mb-4 group"
                  >
                    <Avatar className={`h-10 w-10 flex-shrink-0 ring-2 ring-offset-2 ${isAdmin
                        ? "bg-gradient-to-br from-blue-500 to-blue-600 ring-blue-200 dark:ring-blue-900"
                        : "bg-gradient-to-br from-purple-500 to-purple-600 ring-purple-200 dark:ring-purple-900"
                      }`}>
                      <AvatarFallback className="text-white font-semibold">{isAdmin ? "A" : "S"}</AvatarFallback>
                    </Avatar>
                    <div className="flex-1 max-w-[75%]">
                      <div className="text-xs font-semibold text-gray-700 dark:text-gray-300 mb-1.5">
                        {senderName}
                        {m.is_internal && (
                          <span className="ml-2 text-yellow-600 dark:text-yellow-400 inline-flex items-center gap-1 px-2 py-0.5 bg-yellow-100 dark:bg-yellow-900/30 rounded-full">
                            <AlertCircle className="h-3 w-3" />
                            <span className="text-[10px] font-medium">Internal Note</span>
                          </span>
                        )}
                      </div>
                      <div
                        className={`rounded-2xl px-4 py-3 text-sm shadow-sm hover:shadow-md transition-shadow duration-200 ${m.is_internal
                            ? 'bg-gradient-to-br from-yellow-50 to-yellow-100 dark:from-yellow-900/20 dark:to-yellow-800/20 border-2 border-yellow-300 dark:border-yellow-700'
                            : isAdmin
                              ? 'bg-gradient-to-br from-blue-500 to-blue-600 text-white'
                              : 'bg-white dark:bg-gray-800 text-gray-900 dark:text-gray-100 border border-gray-200 dark:border-gray-700'
                          }`}
                      >
                        {highlight ? (
                          <div
                            className="leading-relaxed"
                            dangerouslySetInnerHTML={{
                              __html: m.message.replace(
                                new RegExp(searchQuery, 'gi'),
                                (match) => `<mark class="bg-yellow-300 dark:bg-yellow-600 px-1 rounded">${match}</mark>`
                              ),
                            }}
                          />
                        ) : (
                          <div className="leading-relaxed">{m.message}</div>
                        )}
                        {m.attachments && m.attachments.length > 0 && (
                          <div className="mt-3 space-y-2">
                            {m.attachments.map((att, idx) => (
                              <button
                                key={idx}
                                onClick={() => handleFileDownload(att.url, att.filename)}
                                className={`flex items-center gap-2 text-xs px-3 py-2 rounded-lg transition-colors w-full text-left ${m.is_internal
                                    ? "bg-yellow-200 dark:bg-yellow-800/30 hover:bg-yellow-300 dark:hover:bg-yellow-700/30"
                                    : isAdmin
                                      ? "bg-blue-600/30 hover:bg-blue-600/50 backdrop-blur-sm"
                                      : "bg-gray-100 dark:bg-gray-700 hover:bg-gray-200 dark:hover:bg-gray-600"
                                  }`}
                              >
                                <Download className="h-3.5 w-3.5 flex-shrink-0" />
                                <span className="truncate">{att.filename}</span>
                              </button>
                            ))}
                          </div>
                        )}
                      </div>
                      <div className="flex items-center gap-1.5 mt-1.5 text-[11px] text-gray-500 dark:text-gray-400">
                        <span className="font-medium">{new Date(m.created_at).toLocaleString()}</span>
                        {!isAdmin && (
                          m.is_read ? (
                            <CheckCheck className="h-3.5 w-3.5 text-blue-500" />
                          ) : (
                            <Check className="h-3.5 w-3.5 text-gray-400" />
                          )
                        )}
                      </div>
                    </div>
                  </motion.div>
                );
              })}
 
              {/* Typing indicator */}
              {remoteTyping.is_typing && (
                <div className="flex gap-2 items-center text-xs text-gray-500 italic">
                  <div className="flex gap-1">
                    <motion.div
                      className="w-2 h-2 bg-gray-400 rounded-full"
                      animate={{ scale: [1, 1.3, 1] }}
                      transition={{ repeat: Infinity, duration: 1, delay: 0 }}
                    />
                    <motion.div
                      className="w-2 h-2 bg-gray-400 rounded-full"
                      animate={{ scale: [1, 1.3, 1] }}
                      transition={{ repeat: Infinity, duration: 1, delay: 0.2 }}
                    />
                    <motion.div
                      className="w-2 h-2 bg-gray-400 rounded-full"
                      animate={{ scale: [1, 1.3, 1] }}
                      transition={{ repeat: Infinity, duration: 1, delay: 0.4 }}
                    />
                  </div>
                  <span>Seller is typing...</span>
                </div>
              )}
            </div>
 
            {/* Composer */}
            <div className="p-3 border-t space-y-2">
              {uploadedFiles.length > 0 && (
                <div className="flex flex-wrap gap-2">
                  {uploadedFiles.map((file, idx) => (
                    <div
                      key={idx}
                      className="flex items-center gap-1 bg-gray-100 dark:bg-gray-800 px-2 py-1 rounded text-xs"
                    >
                      <Paperclip className="h-3 w-3" />
                      <span className="max-w-[150px] truncate">{file.filename}</span>
                      <button
                        onClick={() => setUploadedFiles((prev) => prev.filter((_, i) => i !== idx))}
                      >
                        <X className="h-3 w-3" />
                      </button>
                    </div>
                  ))}
                </div>
              )}
 
              <Tabs value={isInternal ? 'internal' : 'reply'} onValueChange={(v) => setIsInternal(v === 'internal')}>
                <TabsList className="grid w-full grid-cols-2">
                  <TabsTrigger value="reply">Reply to Seller</TabsTrigger>
                  <TabsTrigger value="internal">Internal Note</TabsTrigger>
                </TabsList>
              </Tabs>
 
              <textarea
                rows={3}
                className="w-full px-3 py-2 rounded border bg-transparent resize-none text-sm"
                placeholder={isInternal ? 'Add internal note (only visible to admins)...' : 'Type your reply...'}
                value={reply}
                onChange={(e) => handleReplyChange(e.target.value)}
                onKeyDown={(e) => {
                  if (e.key === 'Enter' && !e.shiftKey) {
                    e.preventDefault();
                    sendReply();
                  }
                }}
              />
 
              <div className="flex items-center justify-between">
                <div className="flex gap-2">
                  <input
                    ref={fileInputRef}
                    type="file"
                    className="hidden"
                    accept=".pdf,.png,.jpg,.jpeg,.gif,.doc,.docx,.txt"
                    onChange={handleFileUpload}
                  />
                  <Button
                    variant="outline"
                    size="sm"
                    onClick={() => fileInputRef.current?.click()}
                    disabled={uploading}
                  >
                    {uploading ? (
                      <Upload className="h-4 w-4 animate-spin" />
                    ) : (
                      <Paperclip className="h-4 w-4" />
                    )}
                  </Button>
                </div>
 
                <Button size="sm" disabled={!reply.trim()} onClick={sendReply}>
                  <Send className="h-4 w-4 mr-1" />
                  {isInternal ? 'Add Note' : 'Send Reply'}
                </Button>
              </div>
            </div>
          </>
        ) : (
          <div className="h-full flex flex-col items-center justify-center text-gray-500">
            <MessageCircle className="h-16 w-16 mb-4 opacity-20" />
            <p className="text-sm">Select a ticket to view conversation</p>
          </div>
        )}
      </div>
    </div>
  );
}
