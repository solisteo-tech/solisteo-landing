interface UsageCardProps {
  title: string;
  current: number;
  limit: number;
  icon: React.ReactNode;
  color: string;
  description?: string;
}
 
function UsageCard({ title, current, limit, icon, color, description }: UsageCardProps) {
  const percentage = limit === -1 ? 0 : Math.min((current / limit) * 100, 100);
  const isUnlimited = limit === -1;
  const isNearLimit = !isUnlimited && percentage > 80;
 
  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.3 }}
    >
      <Card className={`relative ${isNearLimit ? 'border-orange-200 bg-orange-50/50' : ''}`}>
        {isNearLimit && (
          <div className="absolute -top-2 -right-2">
            <Badge variant="destructive" className="text-xs">
              Near Limit
            </Badge>
          </div>
        )}
        <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
          <CardTitle className="text-sm font-medium flex items-center gap-2">
            {icon}
            {title}
          </CardTitle>
          <div className="text-right">
            <div className="text-2xl font-bold">{current}</div>
            <div className="text-xs text-muted-foreground">
              {isUnlimited ? 'Unlimited' : `of ${limit}`}
            </div>
          </div>
        </CardHeader>
        <CardContent>
          {!isUnlimited && (
            <div className="space-y-2">
              <Progress
                value={percentage}
                className={`h-2 ${color}`}
              />
              <div className="flex justify-between text-xs text-muted-foreground">
                <span>{percentage.toFixed(1)}% used</span>
                <span>{limit - current} remaining</span>
              </div>
            </div>
          )}
          {isUnlimited && (
            <div className="flex items-center gap-2 text-sm text-green-600">
              <Zap className="h-4 w-4" />
              Unlimited usage
            </div>
          )}
          {description && (
            <p className="text-xs text-muted-foreground mt-2">{description}</p>
          )}
        </CardContent>
      </Card>
    </motion.div>
  );
}
 
function PlanBadge({ plan }: { plan: SubscriptionPlan }) {
  const getTierConfig = (planName: string) => {
    const lowerName = planName.toLowerCase();
    
    if (lowerName.includes('enterprise') || lowerName.includes('premium')) {
      return {
        icon: <Crown className="h-4 w-4" />,
        color: 'bg-yellow-100 text-yellow-800 border-yellow-200',
        label: plan.name
      };
    } else if (lowerName.includes('growth') || lowerName.includes('pro') || lowerName.includes('priority')) {
      return {
        icon: <Shield className="h-4 w-4" />,
        color: 'bg-blue-100 text-blue-800 border-blue-200',
        label: plan.name
      };
    } else {
      return {
        icon: <Zap className="h-4 w-4" />,
        color: 'bg-gray-100 text-gray-800 border-gray-200',
        label: plan.name
      };
    }
  };
 
  const config = getTierConfig(plan.name);
 
  return (
    <Badge className={`${config.color} flex items-center gap-1 px-3 py-1`}>
      {config.icon}
      {config.label}
    </Badge>
  );
}

  useEffect(() => {
    fetchUsageData();
  }, []);
 
  const fetchUsageData = async () => {
    try {
      const response = await api.get('/api/v1/billing/usage');
      setUsageData(response.data);
    } catch (err: any) {
      setError('Failed to fetch usage data');
      console.error('Failed to fetch usage:', err);
    } finally {
      setIsLoading(false);
    }
  };

    return (
      <div className="space-y-6">
        <div className="animate-pulse">
          <div className="h-8 bg-gray-200 rounded w-1/4 mb-6"></div>

              <div key={i} className="h-32 bg-gray-200 rounded"></div>
            ))}
          </div>
        </div>
