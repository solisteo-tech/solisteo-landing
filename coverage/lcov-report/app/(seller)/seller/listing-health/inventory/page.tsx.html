"use client";
 
import React, { useState } from "react";
import { ChevronLeft, ChevronRight, Info, Search, Filter, X, Calendar } from "lucide-react";
import { Badge } from "@/components/ui/badge";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { Tooltip, TooltipContent, TooltipProvider, TooltipTrigger } from "@/components/ui/tooltip";
import { Popover, PopoverContent, PopoverTrigger } from "@/components/ui/popover";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { cn } from "@/lib/utils";
import { useListingHealthData } from '@/lib/hooks/useListingHealthData';
import { ForceCheckButton } from '@/components/ForceCheckButton';
 
/**
 * ============================================================================
 * AUTOMATION ENDPOINT DOCUMENTATION FOR CHARAN
 * ============================================================================
 * 
 * PRIMARY ENDPOINT: GET /api/v1/seller/listing-health/inventory
 * 
 * Purpose: Fetch inventory items with automated health check data
 * 
 * Query Parameters:
 * - skip: number (pagination offset)
 * - limit: number (items per page, default: 10)
 * - asin: string (optional, filter by ASIN)
 * - msku: string (optional, filter by MSKU)
 * - status: 'live' | 'non_live' | 'all' (filter by health status)
 * - buy_box_status: 'winning' | 'losing' | 'all' (filter by buy box)
 * - date_from: ISO date string (filter scans from this date)
 * - date_to: ISO date string (filter scans until this date)
 * - search: string (search in ASIN/MSKU/Title)
 * 
 * Response Format:
 * {
 *   "data": [
 *     {
 *       "id": string,
 *       "asin": string,
 *       "msku": string,
 *       "title": string,
 *       "health_status": "live" | "non_live",  // From automated scraper
 *       "health_reason": string | null,         // Error reason if non_live
 *       "last_scan_time": ISO datetime,         // When automation last checked
 *       "delivery_times": {                     // From regional scraper
 *         "north": number,                      // Days
 *         "south": number,
 *         "east": number,
 *         "west": number
 *       },
 *       "buy_box": {                           // From SP-API or scraper
 *         "is_owner": boolean,
 *         "winner_name": string,
 *         "price_difference": number           // If not owner
 *       },
 *       "current_price": number,
 *       "last_updated": ISO datetime
 *     }
 *   ],
 *   "total": number,
 *   "skip": number,
 *   "limit": number
 * }
 * 
 * AUTOMATION TASKS:
 * 1. Scheduled Cron Job: Run health checks every 6/12/24 hours
 * 2. Regional Scraper: Check delivery times for all 4 regions
 * 3. Buy Box Monitor: Track buy box winner and pricing
 * 4. Error Handling: Capture scraping errors (Captcha, 404, Timeout, etc.)
 * 
 * ============================================================================
 */
 
// TypeScript interfaces for automation data
interface HealthData {
    id: string;
    asin: string;
    msku: string;
    title: string;
    health_status: 'live' | 'non_live';
    health_reason: string | null;
    last_scan_time: string;
    delivery_times: {
        north: number;
        south: number;
        east: number;
        west: number;
    };
    buy_box: {
        is_owner: boolean;
        winner_name: string;
        price_difference: number;
        winner_deal?: string;
        my_deal?: string;
    };
    current_price: number;
    last_updated: string;
}
 
interface HealthDataResponse {
    data: HealthData[];
    total: number;
    skip: number;
    limit: number;
}
 
export default function InventoryListingsPage() {
    const [page, setPage] = useState(0);
    const pageSize = 10;
 
    // Filter states
    const [searchQuery, setSearchQuery] = useState("");
    const [statusFilter, setStatusFilter] = useState<string>("all");
    const [buyBoxFilter, setBuyBoxFilter] = useState<string>("all");
    const [dateRange, setDateRange] = useState<string>("all");
 
    // Use real listing health data hook
    const { data: healthData, isLoading } = useListingHealthData(
        page * pageSize,
        pageSize,
        searchQuery,
        statusFilter,
        buyBoxFilter,
        undefined, // date_from
        undefined  // date_to
    );
 
    const handleClearFilters = () => {
        setSearchQuery("");
        setStatusFilter("all");
        setBuyBoxFilter("all");
        setDateRange("all");
    };
 
    const hasActiveFilters = searchQuery || statusFilter !== "all" || buyBoxFilter !== "all" || dateRange !== "all";
 
    return (
        <div className="min-h-screen bg-gray-50 p-6 md:p-8 space-y-6 max-w-[1600px] mx-auto">
            {/* Header */}
            <div className="flex flex-col md:flex-row md:items-center md:justify-between space-y-2 md:space-y-0">
                <div>
                    <h1 className="text-2xl font-bold text-gray-900">Your Inventory Listings</h1>
                    <p className="text-gray-500 text-sm">Automated health monitoring for all your inventory items.</p>
                </div>
                <ForceCheckButton />
            </div>
 
            {/* Global Filters */}
            <Card className="border-gray-200 shadow-sm">
                <CardHeader className="border-b border-gray-100 py-4 px-5">
                    <div className="flex items-center justify-between">
                        <div className="flex items-center gap-2">
                            <Filter className="w-4 h-4 text-gray-500" />
                            <CardTitle className="text-base font-semibold">Filters</CardTitle>
                        </div>
                        {hasActiveFilters && (
                            <Button
                                variant="ghost"
                                size="sm"
                                onClick={handleClearFilters}
                                className="h-8 text-xs text-gray-500 hover:text-gray-700"
                            >
                                <X className="w-3 h-3 mr-1" />
                                Clear All
                            </Button>
                        )}
                    </div>
                </CardHeader>
                <CardContent className="p-5">
                    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
                        {/* Search Filter */}
                        <div className="space-y-2">
                            <label className="text-xs font-medium text-gray-700">Search</label>
                            <div className="relative">
                                <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-gray-400" />
                                <Input
                                    placeholder="ASIN, SKU, or Title"
                                    value={searchQuery}
                                    onChange={(e) => setSearchQuery(e.target.value)}
                                    className="pl-9 h-10 text-sm"
                                />
                            </div>
                        </div>
 
                        {/* Status Filter */}
                        <div className="space-y-2">
                            <label className="text-xs font-medium text-gray-700">Health Status</label>
                            <Select value={statusFilter} onValueChange={setStatusFilter}>
                                <SelectTrigger className="h-10">
                                    <SelectValue placeholder="All Status" />
                                </SelectTrigger>
                                <SelectContent>
                                    <SelectItem value="all">All Status</SelectItem>
                                    <SelectItem value="live">Live Only</SelectItem>
                                    <SelectItem value="non_live">Non-Live Only</SelectItem>
                                </SelectContent>
                            </Select>
                        </div>
 
                        {/* Buy Box Filter */}
                        <div className="space-y-2">
                            <label className="text-xs font-medium text-gray-700">Buy Box Status</label>
                            <Select value={buyBoxFilter} onValueChange={setBuyBoxFilter}>
                                <SelectTrigger className="h-10">
                                    <SelectValue placeholder="All" />
                                </SelectTrigger>
                                <SelectContent>
                                    <SelectItem value="all">All</SelectItem>
                                    <SelectItem value="winning">Winning</SelectItem>
                                    <SelectItem value="losing">Losing</SelectItem>
                                </SelectContent>
                            </Select>
                        </div>
 
                        {/* Date Range Filter */}
                        <div className="space-y-2">
                            <label className="text-xs font-medium text-gray-700">Last Scanned</label>
                            <Select value={dateRange} onValueChange={setDateRange}>
                                <SelectTrigger className="h-10">
                                    <SelectValue placeholder="All Time" />
                                </SelectTrigger>
                                <SelectContent>
                                    <SelectItem value="all">All Time</SelectItem>
                                    <SelectItem value="today">Today</SelectItem>
                                    <SelectItem value="week">Last 7 Days</SelectItem>
                                    <SelectItem value="month">Last 30 Days</SelectItem>
                                </SelectContent>
                            </Select>
                        </div>
                    </div>
                </CardContent>
            </Card>
 
            {/* Inventory Table */}
            <Card className="border-gray-200 shadow-sm">
                <CardHeader className="flex flex-row items-center justify-between border-b border-gray-100 py-3 px-4">
                    <div>
                        <CardTitle className="text-base font-semibold">Inventory Health Status</CardTitle>
                        <p className="text-xs text-gray-500 mt-0.5">
                            {healthData?.total || 0} items  Last updated: {new Date().toLocaleTimeString()}
                        </p>
                    </div>
                    <div className="flex items-center gap-1.5">
                        <Button
                            variant="outline"
                            size="sm"
                            className="h-7 w-7 p-0"
                            disabled={page === 0}
                            onClick={() => setPage(p => Math.max(0, p - 1))}
                        >
                            <ChevronLeft className="w-3.5 h-3.5" />
                        </Button>
                        <span className="text-xs text-gray-600 w-14 text-center tabular-nums">
                            Page {page + 1}
                        </span>
                        <Button
                            variant="outline"
                            size="sm"
                            className="h-7 w-7 p-0"
                            disabled={(healthData?.data?.length || 0) < pageSize}
                            onClick={() => setPage(p => p + 1)}
                        >
                            <ChevronRight className="w-3.5 h-3.5" />
                        </Button>
                    </div>
                </CardHeader>
                <CardContent className="p-0">
                    <Table>
                        <TableHeader>
                            <TableRow className="bg-gray-50/50 hover:bg-gray-50/50">
                                <TableHead className="font-semibold text-gray-700 w-[180px]">ASIN / SKU</TableHead>
                                <TableHead className="font-semibold text-gray-700 w-[120px]">Product Title</TableHead>
                                <TableHead className="font-semibold text-gray-700 w-[120px]">Status</TableHead>
                                <TableHead className="font-semibold text-gray-700 text-center" colSpan={4}>Regional Delivery (Days)</TableHead>
                                <TableHead className="font-semibold text-gray-700 w-[180px]">Buy Box Status</TableHead>
                            </TableRow>
                            <TableRow className="bg-white border-b border-gray-100 text-[10px] text-gray-400 uppercase tracking-wider">
                                <TableHead></TableHead>
                                <TableHead></TableHead>
                                <TableHead></TableHead>
                                <TableHead className="font-medium text-center bg-gray-50/30 text-[9px] px-0.5 py-0.5">North</TableHead>
                                <TableHead className="font-medium text-center bg-gray-50/30 text-[9px] px-0.5 py-0.5">South</TableHead>
                                <TableHead className="font-medium text-center bg-gray-50/30 text-[9px] px-0.5 py-0.5">East</TableHead>
                                <TableHead className="font-medium text-center bg-gray-50/30 text-[9px] px-0.5 py-0.5">West</TableHead>
                                <TableHead></TableHead>
                            </TableRow>
                        </TableHeader>
                        <TableBody>
                            {isLoading ? (
                                Array(5).fill(0).map((_, i) => (
                                    <TableRow key={i} className="h-10">
                                        <TableCell className="pl-4"><div className="h-2.5 w-12 bg-gray-100 rounded animate-pulse" /></TableCell>
                                        <TableCell><div className="h-2.5 w-32 bg-gray-100 rounded animate-pulse" /></TableCell>
                                        <TableCell><div className="h-2.5 w-10 bg-gray-100 rounded animate-pulse" /></TableCell>
                                        <TableCell colSpan={4}><div className="h-2.5 w-full bg-gray-100 rounded animate-pulse" /></TableCell>
                                        <TableCell className="pr-4"><div className="h-2.5 w-16 bg-gray-100 rounded animate-pulse" /></TableCell>
                                    </TableRow>
                                ))
                            ) : (healthData?.data || []).map((item: any) => {
                                const renderDelivery = (region: string) => {
                                    const val = item.delivery_times?.[region];
                                    if (typeof val === 'object' && val) {
                                        return (
                                            <div className="flex flex-col items-center leading-none gap-0.5">
                                                <span>{val.days > -1 ? val.days + 'd' : '-'}</span>
                                                <span className="text-[8px] text-gray-400 font-mono">{val.pincode}</span>
                                            </div>
                                        );
                                    }
                                    return (val ?? '-') + 'd';
                                };
                                return (
                                    <TableRow key={item.id} className="hover:bg-blue-50/30 group h-12 border-b border-gray-50 transition-colors">
                                        {/* ASIN / SKU */}
                                        <TableCell className="align-top py-3">
                                            <div className="flex flex-col gap-0.5">
                                                <a
                                                    href={`https://www.amazon.in/dp/${item.asin}`}
                                                    target="_blank"
                                                    rel="noopener noreferrer"
                                                    className="font-mono text-[16px] text-blue-600 hover:text-blue-800 hover:underline transition-colors"
                                                >
                                                    {item.asin}
                                                </a>
                                                <span className="font-mono text-[12px] text-gray-500">{item.msku}</span>
                                            </div>
                                        </TableCell>
 
                                        {/* Title with Tooltip */}
                                        <TableCell className="align-top py-3 max-w-[250px]">
                                            <TooltipProvider>
                                                <Tooltip>
                                                    <TooltipTrigger asChild>
                                                        <div className="text-sm font-medium text-gray-800 truncate cursor-help">
                                                            {item.title}
                                                        </div>
                                                    </TooltipTrigger>
                                                    <TooltipContent className="max-w-[300px] bg-gray-900 text-white text-xs">
                                                        {item.title}
                                                    </TooltipContent>
                                                </Tooltip>
                                            </TooltipProvider>
                                        </TableCell>
 
                                        {/* Health Status */}
                                        <TableCell className="align-center py-3">
                                            <div className="flex items-center gap-2">
                                                <Badge variant="outline" className={cn(
                                                    "text-[10px] uppercase tracking-wide font-bold",
                                                    item.health_status === "live"
                                                        ? "bg-emerald-50 text-emerald-700 border-emerald-200"
                                                        : "bg-red-50 text-red-700 border-red-200"
                                                )}>
                                                    {item.health_status === "live" ? "Live" : "Non Live"}
                                                </Badge>
 
                                                {item.health_status === "non_live" && item.health_reason && (
                                                    <Popover>
                                                        <PopoverTrigger asChild>
                                                            <button className="text-gray-400 hover:text-red-500 transition-colors">
                                                                <Info className="w-4 h-4" />
                                                            </button>
                                                        </PopoverTrigger>
                                                        <PopoverContent className="w-auto p-3 text-xs bg-white border-red-100 shadow-md">
                                                            <div className="font-medium text-red-900 mb-1">Scraping Failed</div>
                                                            <div className="text-red-700">{item.health_reason}</div>
                                                            <div className="text-gray-500 text-[10px] mt-1">
                                                                Last checked: {item.last_scan_time ? new Date(item.last_scan_time).toLocaleString() : 'Never'}
                                                            </div>
                                                        </PopoverContent>
                                                    </Popover>
                                                )}
                                            </div>
                                        </TableCell>
 
                                        {/* Regional Delivery */}
                                        <TableCell className="text-center align-top py-3 text-[10px] font-medium text-gray-600 border-l border-gray-50 w-[50px]">
                                            {renderDelivery('north')}
                                        </TableCell>
                                        <TableCell className="text-center align-top py-3 text-[10px] font-medium text-gray-600 border-l border-gray-50 w-[50px]">
                                            {renderDelivery('south')}
                                        </TableCell>
                                        <TableCell className="text-center align-top py-3 text-[10px] font-medium text-gray-600 border-l border-gray-50 w-[50px]">
                                            {renderDelivery('east')}
                                        </TableCell>
                                        <TableCell className="text-center align-top py-3 text-[10px] font-medium text-gray-600 border-l border-gray-50 border-r w-[50px]">
                                            {renderDelivery('west')}
                                        </TableCell>
 
                                        {/* Buy Box Status */}
                                        <TableCell className="align-top py-3">
                                            <div className="flex flex-col gap-1">
                                                {item.buy_box?.is_owner ? (
                                                    <>
                                                        <div className="flex items-center gap-1.5 text-emerald-700 bg-emerald-50 px-2 py-1 rounded w-fit">
                                                            <div className="w-1.5 h-1.5 rounded-full bg-emerald-600"></div>
                                                            <span className="text-xs font-bold">You (Winning)</span>
                                                        </div>
                                                        {item.buy_box?.my_deal && (
                                                            <span className="text-[10px] font-bold text-emerald-600 border border-emerald-200 bg-white px-1.5 py-0.5 rounded ml-1">
                                                                {item.buy_box.my_deal}
                                                            </span>
                                                        )}
                                                    </>
                                                ) : (
                                                    <div className="space-y-0.5">
                                                        <div className="text-xs font-medium text-gray-500">
                                                            Winner: <span className="text-gray-900">{item.buy_box?.winner_name || 'Unknown'}</span>
                                                        </div>
                                                        {item.buy_box?.price_difference > 0 && (
                                                            <div className="text-xs font-bold text-red-600 bg-red-50 px-1.5 py-0.5 rounded w-fit">
                                                                +{item.buy_box.price_difference.toFixed(2)} gap
                                                            </div>
                                                        )}
                                                        {item.buy_box?.winner_deal && (
                                                            <div className="text-[10px] font-bold text-red-600 border border-red-200 bg-red-50 px-1.5 py-0.5 rounded w-fit mt-0.5">
                                                                {item.buy_box.winner_deal}
                                                            </div>
                                                        )}
                                                        {item.buy_box?.my_deal && (
                                                            <div className="text-[10px] text-gray-500 mt-0.5">
                                                                Your Offer: <span className="font-medium text-emerald-600">{item.buy_box.my_deal}</span>
                                                            </div>
                                                        )}
                                                    </div>
                                                )}
                                            </div>
                                        </TableCell>
                                    </TableRow>
                                );
                            })}
                        </TableBody>
                    </Table>
                </CardContent>
            </Card>
 
            {/* Info Banner */}
            <Card className="border-blue-200 bg-blue-50">
                <CardContent className="p-4">
                    <div className="flex gap-3 items-start">
                        <Info className="w-5 h-5 text-blue-600 mt-0.5 flex-shrink-0" />
                        <div className="text-sm text-blue-900">
                            <p className="font-semibold mb-1">Automation Status</p>
                            <p className="text-blue-700">
                                Health checks run automatically every 6 hours. Regional delivery times and buy box data are updated in real-time through our scraping infrastructure.
                            </p>
                        </div>
                    </div>
                </CardContent>
            </Card>
        </div>
    );
}
